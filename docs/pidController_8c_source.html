126 double pidHandler(int currentPosition){
127  if(DEBUG) printf("[DEBUG] - Starting pidHandler pidController \n");
128  double pidUnitVoltage;
129  double pPart;
130  double iPart;
131  double dPart;
132  double deviation;
133  double tempErrorDeviation = targetLocation - currentPosition;
134 
135  allErrorsSummation += tempErrorDeviation;
136  deviation = tempErrorDeviation - errorDeviation;
137  errorDeviation = tempErrorDeviation;
138  pPart = controlKp * tempErrorDeviation;
139  iPart = controlKi * allErrorsSummation;
140  dPart = controlKd * deviation;
141 
142  pidUnitVoltage = pPart + iPart + dPart;
143 
144  if(pidUnitVoltage > QSER_VOLTAGE) pidUnitVoltage = QSER_VOLTAGE;
145  if(pidUnitVoltage < -QSER_VOLTAGE) pidUnitVoltage = -QSER_VOLTAGE;
146 
147  printf("[INFO] - PID Voltage Result %f \n", pidUnitVoltage);
148  return pidUnitVoltage;
149 }
