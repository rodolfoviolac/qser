124 double pidHandler(int currentPosition){
125  if(DEBUG) printf("[DEBUG] - Starting pidHandler pidControler \n");
126  double pidUnitVoltage;
127  double pPart;
128  double iPart;
129  double dPart;
130  double deviation;
131  double tempErrorDeviation = targetLocation - currentPosition;
132 
133  allErrorsSummation += tempErrorDeviation;
134  deviation = tempErrorDeviation - errorDeviation;
135  errorDeviation = tempErrorDeviation;
136  pPart = controlKp * tempErrorDeviation;
137  iPart = controlKi * allErrorsSummation;
138  dPart = controlKd * deviation;
139 
140  pidUnitVoltage = pPart + iPart + dPart;
141 
142  if(pidUnitVoltage > QSER_VOLTAGE) pidUnitVoltage = QSER_VOLTAGE;
143  if(pidUnitVoltage < -QSER_VOLTAGE) pidUnitVoltage = -QSER_VOLTAGE;
144 
145  printf("[INFO] - PID Voltage Result %f \n", pidUnitVoltage);
146  return pidUnitVoltage;
147 }
