40 void* handlerMainThread() {

42  printUserGroupsName();

45  char command[100];
46  while (fgets(command, sizeof(command), stdin)) {
47  if(DEBUG) printf("[DEBUG] - Inside Command While MainThread \n");
48  switch (resolveOption(command)) {
49  case OPTION_SPEED: {
50  if(DEBUG) printf("[DEBUG] - Got SPEED Option MainThread \n");
51  float commandAmount = atof(&command[6]);
52  if(commandAmount > 100 || commandAmount < 0) {
53  printf("[ERROR] - Speed Amount Must Be [0-100]\n");
54  } else {
55  if(DEBUG) printf("[DEBUG] - Got Valid Speed Number MainThread \n");
56  float dcAmount = (commandAmount/100)*atof(PWM_PERIOD);
57  if(handlePwm((int)dcAmount) == 1) {
58  fprintf(stdout, "[INFO] MT - Motor Duty Cycle %d%%\n\n", (int)dcAmount/10000);
59  } else {
60  printf("[ERROR] MT - Error Turning On Motor\n\n");
61  }
62  }
63  break;
64  }
65  case OPTION_JPL: {
66  if(DEBUG) printf("[DEBUG] - Got JPL Option MainThread \n");
67  if(JPL_REACHED_STATUS_ONE == JPL_REACHED){
68  printf("[INFO] MT - JPL ONE Reached The End. State: Active [ %d ]\n", JPL_REACHED_STATUS_ONE);
69  } else printf("[INFO] MT - READ JPL ONE. State: Inactive [ %d ]\n", JPL_REACHED_STATUS_ONE);
70  if(JPL_REACHED_STATUS_TWO == JPL_REACHED){
71  printf("[INFO] MT - JPL TWO Reached The End. State: Active [ %d ]\n\n", JPL_REACHED_STATUS_TWO);
72  } else printf("[INFO] MT - READ JPL TWO. State: Inactive [ %d ]\n\n", JPL_REACHED_STATUS_TWO);
73  break;
74  }
75  case OPTION_GO_FOR_V: {
76  if(DEBUG) printf("[DEBUG] - Got GO FOR V Option MainThread \n");
77  float commandAmount = atof(&command[7]);
78  if(commandAmount < -QSER_VOLTAGE || commandAmount > QSER_VOLTAGE) {
79  fprintf(stderr, "[ERROR] - Voltage Must Be On The Interval [%+d,%+d]\n", (int) -QSER_VOLTAGE, (int) QSER_VOLTAGE);
80  } else {
81  if(DEBUG) printf("[DEBUG] MT - Got Accepted Voltage \n");
82  if(qserVoltage(commandAmount) == 1) {
83  fprintf(stdout, "[INFO] - Motor Triggered With Voltage %+.1fV\n\n", commandAmount);
84  } else {
85  fprintf(stdout, "[ERROR] - Something Went Wrong Trying To Turn On Motor\n\n");
86  }
87  }
88  break;
89  }
90  case OPTION_QSER_BREAK: {
91  if(DEBUG) printf("[DEBUG] - Got QSER BREAK Option MainThread\n");
92  if(qserBreak() == TRUE){
93  turnOffPid();
94  printf("[INFO] - Breaking Motor\n");
95  } else {
96  fprintf(stdout, "[ERROR] - Something Went Wrong Trying To Break Motor\n\n");
97  }
98  break;
99  }
100  case OPTION_SHIFT_RAD: {
101  if(DEBUG) printf("[DEBUG] - Got OPTION_SHIFT_RAD Option MainThread\n");
102  double commandAmount = (double)strtof(&command[9], NULL);
103  if(commandAmount < 0 || commandAmount > 2*PI){
104  printf("[Error] - Radian Must Be On The Interval [0, %f] \n", 2*PI);
105  break;
106  }
107  setTarget((int)(commandAmount*RAD2G*conversionCoefficientUnit));

109  break;
110  }
111  case OPTION_SET_KD: {
112  if(DEBUG) printf("[DEBUG] - Got OPTION_SET_KD Option MainThread \n");
113  double commandAmount = (double)strtof(&command[6], NULL);
114  setControlKd(commandAmount);
115  break;
116  }
117  case OPTION_SET_KP: {
118  if(DEBUG) printf("[DEBUG] - Got OPTION_SET_KP Option MainThread \n");

134  }

137  clearDecoder();
138  printf("[INFO] - Cleaning Decoder\n\n");
139  break;
140  }
141  case OPTION_READ_DEC: {
142  if(DEBUG) printf("[DEBUG] - Got OPTION_READ_DEC Option MainThread\n");
143  int counterDecoder = readDecoderCounter();
144  printf("[INFO] - Decoder Reading: %d \n\n", counterDecoder);
145  break;
146  }
147  case OPTION_READ_DEC_RAD: {
148  if(DEBUG) printf("[DEBUG] - Got OPTION_READ_DEC_RAD Option MainThread\n");
149  int counterDecoder = readDecoderCounter();
150  double radDec = (counterDecoder / conversionCoefficientUnit) * CONST_UNIT_RAD_MATH;
151  printf("[INFO] - Decoder Reading: %f \n\n", radDec);
152  break;
153  }
154  case OPTION_HELP: {
155  if(DEBUG) printf("[DEBUG] - Got HELP Option MainThread\n");
156  printUserOptions();
157  break;
158  }
159  case OPTION_EXIT: {
160  if(DEBUG) printf("[DEBUG] - Got EXIT Option MainThread\n");
161  qserBreak();
162  turnOffPid();
163  wrapUpDecoder();
164  printf("[INFO] - Finishing Application...\n");
165  exit(0);
166  }
167  case OPTION_INVALID: {
168  if(DEBUG) printf("[DEBUG] - Got INVALID Option MainThread\n");
169  printf("[ERROR] - Command Invalid Type Help For The Command List\n");
170  break;
171  }
172  default: {
173  if(DEBUG) printf("[DEBUG] - Got DEFAULT Option MainThread\n");
174  printf("[ERROR] - Command Invalid Type Help For The Command List\n");
175  }
176  }
177  }
178  return NULL;
179 }
